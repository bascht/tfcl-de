describe 'Think first, Code later' do

  let(:product) { double(Product, name: 'Fancy Product Name') }

  it 'should help building good software' do
    while product.open_tickets?
      Developers.should_receive :requirements

      Developers.think_about_problem! and Developers.talk_to_stakeholder!

      Developers.write_tests
      Developers.implement_features or Developers.refactor_code
      Developers.release product.new_version
    end
  end

end
